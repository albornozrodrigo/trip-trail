    ```typescript
    import { NextRequest, NextResponse } from 'next/server'
    import { randomUUID } from 'crypto'
    import { createServerSupabaseClient } from '@/lib/supabaseServer'
    
    export async function POST(request: NextRequest) {
      try {
        const body = await request.json()
        const { tripId } = body as { tripId: string }
        
        if (!tripId) {
          return NextResponse.json(
            { error: 'Trip ID required' },
            { status: 400 }
          )
        }
        
        // Verify user owns this trip
        const supabase = createServerSupabaseClient()
        const { data: { user } } = await supabase.auth.getUser()
        if (!user) {
          return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })
        }
        
        const { data: trip } = await supabase
          .from('trips')
          .select('id')
          .eq('id', tripId)
          .eq('user_id', user.id)
          .single()
        
        if (!trip) {
          return NextResponse.json(
            { error: 'Trip not found or unauthorized' },
            { status: 403 }
          )
        }
        
        // Generate unique token
        const token = randomUUID()
        
        // Create share link
        const { data: shareLink } = await supabase
          .from('share_links')
          .insert({
            trip_id: tripId,
            token
          })
          .select()
          .single()
        
        const shareUrl = `${process.env.NEXT_PUBLIC_BASE_URL}/share/${token}`
        
        return NextResponse.json({
          shareUrl,
          token,
          createdAt: shareLink.created_at
        })
      } catch (error) {
        console.error('Share link creation failed:', error)
        return NextResponse.json(
          { error: 'Failed to create share link' },
          { status: 500 }
        )
      }
    }
    ```
    
    Environment variables required (add to `.env.local`):
    ```
    NEXT_PUBLIC_BASE_URL=https://your-domain.vercel.app
    ```

    ```typescript
    import { createServerSupabaseClient } from '@/lib/supabaseServer'
    import { notFound } from 'next/navigation'
    import ItineraryDisplay from '@/components/ItineraryDisplay'
    
    interface SharePageProps {
      params: { token: string }
    }
    
    export default async function SharePage({ params }: SharePageProps) {
      const supabase = createServerSupabaseClient()
      
      // Lookup share link
      const { data: shareLink } = await supabase
        .from('share_links')
        .select('trip_id')
        .eq('token', params.token)
        .single()
      
      if (!shareLink) {
        notFound()
      }
      
      // Fetch trip and itinerary
      const { data: trip } = await supabase
        .from('trips')
        .select('*')
        .eq('id', shareLink.trip_id)
        .single()
      
      const { data: itineraries } = await supabase
        .from('itineraries')
        .select('*')
        .eq('trip_id', shareLink.trip_id)
        .order('day_number', { ascending: true })
        .order('activity_order', { ascending: true })
      
      if (!trip) {
        notFound()
      }
      
      // Group itineraries by day
      const days = new Map<number, typeof itineraries>()
      itineraries?.forEach(item => {
        if (!days.has(item.day_number)) {
          days.set(item.day_number, [])
        }
        days.get(item.day_number)?.push(item)
      })
      
      const formattedItinerary = Array.from(days.entries()).map(([dayNum, activities]) => ({
        date: trip.date_from,
        dayNumber: dayNum,
        activities: activities.map(a => ({
          time: `${a.time_start} - ${a.time_end}`,
          name: a.activity_name,
          location: a.location || '',
          description: a.description || ''
        }))
      }))
      
      return (
        <div className="min-h-screen bg-gradient-to-br from-blue-50 to-indigo-100 py-8">
          <div className="max-w-4xl mx-auto p-6 bg-white rounded-lg shadow">
            <div className="mb-6 pb-6 border-b">
              <h1 className="text-3xl font-bold mb-2">{trip.destination}</h1>
              <p className="text-gray-600">
                {trip.date_from} to {trip.date_to} • {trip.num_travelers} traveler(s)
              </p>
              <p className="text-sm text-gray-500 mt-2">Shared itinerary (view-only)</p>
            </div>
            
            <div className="space-y-6">
              {formattedItinerary.map((day) => (
                <div key={day.dayNumber} className="border-l-4 border-blue-500 pl-6 py-4">
                  <h2 className="text-2xl font-bold mb-4">Day {day.dayNumber}</h2>
                  <p className="text-sm text-gray-500 mb-4">{day.date}</p>
                  
                  <div className="space-y-4">
                    {day.activities.map((activity, idx) => (
                      <div key={idx} className="bg-gray-50 p-4 rounded">
                        <div className="flex items-start justify-between mb-2">
                          <h3 className="text-lg font-semibold">{activity.name}</h3>
                          <span className="text-sm font-medium text-blue-600">{activity.time}</span>
                        </div>
                        <p className="text-sm text-gray-600 mb-2">{activity.location}</p>
                        <p className="text-gray-700">{activity.description}</p>
                      </div>
                    ))}
                  </div>
                </div>
              ))}
            </div>
          </div>
        </div>
      )
    }
    ```

    ```typescript
    'use client'
    import { useState } from 'react'
    import { useTripStore } from '@/stores/tripStore'
    
    export default function ShareLink() {
      const { currentItinerary, loading } = useTripStore()
      const [shareUrl, setShareUrl] = useState<string | null>(null)
      const [generating, setGenerating] = useState(false)
      const [copied, setCopied] = useState(false)
      
      const handleGenerateShare = async () => {
        if (!currentItinerary) return
        
        setGenerating(true)
        try {
          const response = await fetch('/api/share', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ tripId: currentItinerary.tripId })
          })
          
          if (!response.ok) throw new Error('Failed to generate share link')
          
          const { shareUrl } = await response.json()
          setShareUrl(shareUrl)
        } catch (error) {
          console.error('Share generation failed:', error)
          alert('Failed to generate share link')
        } finally {
          setGenerating(false)
        }
      }
      
      const handleCopyToClipboard = () => {
        if (!shareUrl) return
        navigator.clipboard.writeText(shareUrl)
        setCopied(true)
        setTimeout(() => setCopied(false), 2000)
      }
      
      if (!currentItinerary) return null
      
      return (
        <div className="mt-8 pt-8 border-t">
          <h2 className="text-2xl font-bold mb-4">Share This Trip</h2>
          
          {!shareUrl ? (
            <button
              onClick={handleGenerateShare}
              disabled={generating || loading}
              className="w-full bg-green-600 text-white font-bold py-3 rounded hover:bg-green-700 disabled:bg-gray-400"
            >
              {generating ? 'Generating...' : 'Generate Share Link'}
            </button>
          ) : (
            <div className="bg-blue-50 p-4 rounded border border-blue-200">
              <p className="text-sm text-gray-600 mb-2">Share this link with others:</p>
              <div className="flex gap-2">
                <input
                  type="text"
                  value={shareUrl}
                  readOnly
                  className="flex-1 p-2 border rounded bg-white"
                />
                <button
                  onClick={handleCopyToClipboard}
                  className="bg-blue-600 text-white px-4 py-2 rounded hover:bg-blue-700"
                >
                  {copied ? '✓ Copied' : 'Copy'}
                </button>
              </div>
              <p className="text-xs text-gray-500 mt-2">
                Anyone with this link can view (but not edit) your itinerary.
              </p>
            </div>
          )}
        </div>
      )
    }
    ```
