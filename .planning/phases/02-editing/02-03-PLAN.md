    import { useRef } from 'react'
    
    export function useAutoSave() {
      const timeoutRef = useRef<ReturnType<typeof setTimeout> | null>(null)
      const isSavingRef = useRef(false)
    
      const debounce = (
        callback: () => Promise<void> | void,
        delay: number = 1000
      ) => {
        // Clear previous timeout
        if (timeoutRef.current) {
          clearTimeout(timeoutRef.current)
        }
    
        // Set new timeout
        timeoutRef.current = setTimeout(async () => {
          if (isSavingRef.current) return
    
          isSavingRef.current = true
          try {
            await callback()
          } catch (error) {
            console.error('Auto-save failed:', error)
            // TODO: Show toast notification to user
          } finally {
            isSavingRef.current = false
          }
        }, delay)
      }
    
      const cancel = () => {
        if (timeoutRef.current) {
          clearTimeout(timeoutRef.current)
        }
      }
    
      return { debounce, cancel }
    }
    ```

    'use client'
    import { useEffect, useRef, useState } from 'react'
    import { mapboxgl, getOptimizedRoute, geocodeLocation } from '@/lib/mapbox'
    
    interface Activity {
      activity_name: string
      location: string
      time_start: string
      time_end: string
    }
    
    interface ItineraryMapProps {
      activities: Activity[]
      destination: string
    }
    
    export default function ItineraryMap({
      activities,
      destination
    }: ItineraryMapProps) {
      const mapContainer = useRef<HTMLDivElement>(null)
      const map = useRef<mapboxgl.Map | null>(null)
      const [loading, setLoading] = useState(true)
      const [error, setError] = useState<string | null>(null)
    
      useEffect(() => {
        if (!mapContainer.current) return
    
        const initMap = async () => {
          try {
            setLoading(true)
            setError(null)
    
            // Initialize map at destination center
            const [destLng, destLat] = await geocodeLocation(destination)
    
            map.current = new mapboxgl.Map({
              container: mapContainer.current,
              style: 'mapbox://styles/mapbox/streets-v12',
              center: [destLng, destLat],
              zoom: 12
            })
    
            // Add markers and route
            if (activities.length > 0) {
              const routePoints = []
              const markers: any[] = []
    
              // Geocode activities
              for (let i = 0; i < activities.length; i++) {
                try {
                  const [lng, lat] = await geocodeLocation(activities[i].location)
                  routePoints.push({
                    longitude: lng,
                    latitude: lat,
                    name: activities[i].activity_name
                  })
    
                  // Create marker
                  const marker = document.createElement('div')
                  marker.className = 'w-8 h-8 bg-blue-600 rounded-full flex items-center justify-center text-white font-bold text-sm'
                  marker.innerHTML = String(i + 1)
    
                  new mapboxgl.Marker({ element: marker })
                    .setLngLat([lng, lat])
                    .addTo(map.current!)
                } catch (err) {
                  console.warn(`Could not geocode: ${activities[i].location}`)
                }
              }
    
              // Draw route
              if (routePoints.length > 1) {
                try {
                  const route = await getOptimizedRoute(routePoints)
    
                  const source = {
                    type: 'geojson' as const,
                    data: {
                      type: 'Feature' as const,
                      geometry: route.geometry,
                      properties: {}
                    }
                  }
    
                  map.current!.addSource('route', source)
                  map.current!.addLayer({
                    id: 'route',
                    type: 'line',
                    source: 'route',
                    layout: {
                      'line-join': 'round',
                      'line-cap': 'round'
                    },
                    paint: {
                      'line-color': '#3B82F6',
                      'line-width': 3
                    }
                  })
    
                  // Fit bounds to route
                  const bounds = new mapboxgl.LngLatBounds()
                  routePoints.forEach(point => {
                    bounds.extend([point.longitude, point.latitude])
                  })
                  map.current!.fitBounds(bounds, { padding: 50 })
                } catch (err) {
                  console.warn('Could not draw route:', err)
                }
              }
            }
    
            setLoading(false)
          } catch (err) {
            setError((err as Error).message)
            setLoading(false)
          }
        }
    
        initMap()
    
        return () => {
          if (map.current) map.current.remove()
        }
      }, [activities, destination])
    
      return (
        <div className="bg-gray-100 rounded-lg overflow-hidden h-96">
          {error && (
            <div className="absolute inset-0 flex items-center justify-center bg-red-50 text-red-600">
              Map error: {error}
            </div>
          )}
          {loading && (
            <div className="absolute inset-0 flex items-center justify-center bg-gray-200">
              Loading map...
            </div>
          )}
          <div ref={mapContainer} className="w-full h-full" />
        </div>
      )
    }
    ```

    import { NextRequest, NextResponse } from 'next/server'
    import { createServerSupabaseClient } from '@/lib/supabaseServer'
    
    export async function PATCH(
      request: NextRequest,
      { params }: { params: { tripId: string } }
    ) {
      try {
        const body = await request.json()
        const { activities } = body as {
          activities: Array<{ id: string; dayNumber: number; order: number }>
        }
        
        if (!activities || !Array.isArray(activities)) {
          return NextResponse.json(
            { error: 'activities array required' },
            { status: 400 }
          )
        }
        
        // Verify user owns trip
        const supabase = createServerSupabaseClient()
        const { data: { user } } = await supabase.auth.getUser()
        if (!user) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })
        
        const { data: trip } = await supabase
          .from('trips')
          .select('id')
          .eq('id', params.tripId)
          .eq('user_id', user.id)
          .single()
        
        if (!trip) return NextResponse.json({ error: 'Trip not found' }, { status: 403 })
        
        // Update activity orders in batch
        for (const activity of activities) {
          await supabase
            .from('itineraries')
            .update({
              activity_order: activity.order,
              updated_at: new Date().toISOString()
            })
            .eq('id', activity.id)
            .eq('trip_id', params.tripId)
        }
        
        return NextResponse.json({ message: 'Itinerary updated' })
      } catch (error) {
        console.error('Itinerary update failed:', error)
        return NextResponse.json(
          { error: 'Failed to update itinerary' },
          { status: 500 }
        )
      }
    }
