    ```sql
    -- Add columns for soft delete and custom activity tracking
    ALTER TABLE itineraries ADD COLUMN IF NOT EXISTS deleted_at TIMESTAMP NULL;
    ALTER TABLE itineraries ADD COLUMN IF NOT EXISTS custom_activity BOOLEAN DEFAULT FALSE;
    ALTER TABLE trips ADD COLUMN IF NOT EXISTS last_edited_at TIMESTAMP DEFAULT NOW();
    
    -- Add indexes for performance
    CREATE INDEX IF NOT EXISTS idx_itineraries_deleted_at ON itineraries(deleted_at);
    CREATE INDEX IF NOT EXISTS idx_trips_last_edited_at ON trips(last_edited_at);
    
    -- Create function to update trip timestamp on activity changes
    CREATE OR REPLACE FUNCTION public.update_trip_last_edited()
    RETURNS TRIGGER AS $$
    BEGIN
      UPDATE public.trips SET last_edited_at = NOW() WHERE id = NEW.trip_id;
      RETURN NEW;
    END;
    $$ LANGUAGE plpgsql;
    
    -- Drop trigger if exists (to prevent conflicts)
    DROP TRIGGER IF EXISTS trigger_update_trip_last_edited ON public.itineraries;
    
    -- Create trigger to call function
    CREATE TRIGGER trigger_update_trip_last_edited
    AFTER INSERT, UPDATE, DELETE ON public.itineraries
    FOR EACH ROW EXECUTE FUNCTION public.update_trip_last_edited();
    
    -- Update RLS policies to respect soft deletes
    CREATE POLICY "Users can view own itineraries (not deleted)"
      ON public.itineraries FOR SELECT
      USING (
        trip_id IN (
          SELECT id FROM public.trips WHERE user_id = auth.uid()
        ) AND deleted_at IS NULL
      );
    ```

       ```typescript
       // Configure drop animation
       const dropAnimation: DropAnimation = {
         sideEffects: defaultDropAnimationSideEffects({
           styles: {
             active: {
               opacity: '0.4'
             }
           }
         })
       }
       
       // Configure sensors (pointer, touch, keyboard)
       export const DND_KIT_SENSORS = [
         PointerSensor,
         TouchSensor,
         KeyboardSensor
       ]
       
       // Helper: reorder array after drag
       export const handleDragEnd = (event: DragEndEvent, items: any[]) => {
         const { active, over } = event
         
         if (over && active.id !== over.id) {
           const oldIndex = items.findIndex(item => item.id === active.id)
           const newIndex = items.findIndex(item => item.id === over.id)
           return arrayMove(items, oldIndex, newIndex)
         }
         
         return items
       }
       
       export { DndContext, closestCenter, arrayMove, dropAnimation }
       ```

    5. Create `src/lib/mapbox.ts`:
       ```typescript
       import mapboxgl from 'mapbox-gl'
       import 'mapbox-gl/dist/mapbox-gl.css'
       
       // Initialize Mapbox
       mapboxgl.accessToken = process.env.NEXT_PUBLIC_MAPBOX_TOKEN || ''
       
       export interface RoutePoint {
         longitude: number
         latitude: number
         name: string
       }
       
       export interface OptimizedRoute {
         geometry: { coordinates: [number, number][] }
         distance: number
         duration: number
       }
       
       // Get coordinates for location (using Mapbox Geocoding)
       export async function geocodeLocation(location: string): Promise<[number, number]> {
         const response = await fetch(
           `https://api.mapbox.com/geocoding/v5/mapbox.places/${encodeURIComponent(location)}.json?access_token=${mapboxgl.accessToken}`
         )
         const data = await response.json()
         
         if (!data.features.length) {
           throw new Error(`Location not found: ${location}`)
         }
         
         return data.features[0].center as [number, number]
       }
       
       // Get optimized route between points
       export async function getOptimizedRoute(points: RoutePoint[]): Promise<OptimizedRoute> {
         if (points.length < 2) {
           throw new Error('Need at least 2 points for a route')
         }
         
         const coordinates = points
           .map(p => `${p.longitude},${p.latitude}`)
           .join(';')
         
         const response = await fetch(
           `https://api.mapbox.com/directions/v5/mapbox/walking/${coordinates}?access_token=${mapboxgl.accessToken}&geometries=geojson`
         )
         
         if (!response.ok) {
           throw new Error('Failed to get route')
         }
         
         const data = await response.json()
         return {
           geometry: data.routes[0].geometry,
           distance: data.routes[0].distance,
           duration: data.routes[0].duration
         }
       }
       
       export { mapboxgl }
       ```
    
    6. Verify:
       ```bash
       npm list mapbox-gl @types/mapbox-gl
       ```
